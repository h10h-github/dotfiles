def json_regexp
  str = regexp.inspect.
        sub('\A' , '^').
        sub('\Z' , '$').
        sub('\z' , '$').
        sub(/^\// , '').
        sub(/\/[a-z]*$/ , '').
        gsub(/\(\?#.+\)/ , '').
        gsub(/\(\?-\w+:/ , '(').
        gsub(/\s/ , '')
  Regexp.new(str).source
end
