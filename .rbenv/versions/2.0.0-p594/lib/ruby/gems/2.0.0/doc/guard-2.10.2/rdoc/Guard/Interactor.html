def convert_scope(entries)
  scopes  = { plugins: [], groups: [] }
  unknown = []

  session = Guard.state.session

  entries.each do |entry|
    if plugin = session.plugins.all(entry).first
      scopes[:plugins] << plugin
    elsif group = session.groups.all(entry).first
      scopes[:groups] << group
    else
      unknown << entry
    end
  end

  [scopes, unknown]
end
