def notify(message, opts = {})
  super

  if opts[:path]
    format = opts.fetch(:format, DEFAULTS[:format])

    _write(opts[:path], format % [opts[:type], opts[:title], message])
  else
    ::Guard::UI.error ":file notifier requires a :path option"
  end
end
