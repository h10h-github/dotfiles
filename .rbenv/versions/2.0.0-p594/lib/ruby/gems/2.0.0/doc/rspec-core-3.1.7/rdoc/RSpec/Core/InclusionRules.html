def apply_standalone_filter(updated)
  return true if standalone?
  return nil unless is_standalone_filter?(updated)

  replace_filters(updated)
  true
end

def is_standalone_filter?(rules)
  STANDALONE_FILTERS.any? { |key| rules.key?(key) }
end
