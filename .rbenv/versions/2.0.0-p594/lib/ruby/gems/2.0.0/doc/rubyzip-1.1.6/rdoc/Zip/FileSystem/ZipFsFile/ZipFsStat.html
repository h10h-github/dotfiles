def ftype
  if file?
    return "file"
  elsif directory?
    return "directory"
  else
    raise StandardError, "Unknown file type"
  end
end

def mode
  e = get_entry
  if e.fstype == 3
    e.external_file_attributes >> 16
  else
    33206 # 33206 is equivalent to -rw-rw-rw-
  end
end
