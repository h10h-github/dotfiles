def new_http_client
  proxy = nil

  if @proxy
    unless @proxy.respond_to?(:http) && url = @proxy.http
      raise Error::WebDriverError, "expected HTTP proxy, got #{@proxy.inspect}"
    end
    proxy = URI.parse(url)
  end

  Net::HTTP::Persistent.new "webdriver", proxy
end
