def handshake_keys
  keys = [
    ['Upgrade', 'websocket'],
    ['Connection', 'Upgrade']
  ]
  host = @handshake.host
  host += ":#{@handshake.port}" if @handshake.port
  keys << ['Host', host]
  keys += super
  keys << ['Sec-WebSocket-Origin', @handshake.origin] if @handshake.origin
  keys << ['Sec-WebSocket-Version', @handshake.version]
  keys << ['Sec-WebSocket-Key', key]
  keys
end
