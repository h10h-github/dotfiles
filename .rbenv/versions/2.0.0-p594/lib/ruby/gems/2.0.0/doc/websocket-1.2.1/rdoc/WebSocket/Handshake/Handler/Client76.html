def generate_key(key)
  spaces = 1 + rand(12)
  max = 0xffffffff / spaces
  number = rand(max + 1)
  instance_variable_set("@#{key}_number", number)
  key = (number * spaces).to_s
  (1 + rand(12)).times() do
    char = NOISE_CHARS[rand(NOISE_CHARS.size)]
    pos = rand(key.size + 1)
    key[pos...pos] = char
  end
  spaces.times() do
    pos = 1 + rand(key.size - 1)
    key[pos...pos] = ' '
  end
  return key
end
