def []=(sql, key)
  while @max <= cache.size
    dealloc(cache.shift.last)
  end
  @counter += 1
  cache[sql] = key
end

def clear
  cache.each_value do |stmt_key|
    dealloc stmt_key
  end
  cache.clear
end
