def execute
  if options[:quiet]
    Compass::Frameworks::ALL.each do |framework|
      puts framework.name unless framework.name =~ /^_/
    end
  else
    puts "Available Frameworks & Patterns:\n\n"
    puts Compass::Frameworks.pretty_print
  end
end
