    module HTTP::Headers::Mixin

    Provides shared behavior for {HTTP::Request} and {HTTP::Response}. Expects `@headers` to be an instance of {HTTP::Headers}.

    @example Usage

    class MyHttpRequest
  include HTTP::Headers::Mixin

  def initialize
    @headers = HTTP::Headers.new
  end
end
