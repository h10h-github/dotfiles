# @return [String] eagerly consume the entire body as a string
def to_s
  return @contents if @contents
  fail StateError, 'body is being streamed' unless @streaming.nil?

  begin
    @streaming = false
    @contents = ''
    while (chunk = @client.readpartial)
      @contents << chunk
    end
  rescue
    @contents = nil
    raise
  end
