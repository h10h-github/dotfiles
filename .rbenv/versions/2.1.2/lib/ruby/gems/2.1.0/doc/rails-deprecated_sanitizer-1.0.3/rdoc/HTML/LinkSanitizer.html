def sanitizeable?(text)
  !(text.nil? || text.empty? || !((text.index("<a") || text.index("<href")) && text.index(">")))
end

def process_node(node, result, options)
  result << node.to_s unless node.is_a?(HTML::Tag) && included_tags.include?(node.name)
end
