def distance_in_words(now = Time.now)
  distance = now - @start

  if distance < MINUTE
    pluralize(distance, "sec")
  elsif distance < HOUR
    pluralize(distance / MINUTE, "min")
  else
    pluralize(distance / HOUR, "hour")
  end
end

def interval
  distance = Time.now - @start

  if distance < MINUTE
    SECOND
  elsif distance < HOUR
    MINUTE
  else
    HOUR
  end
end
