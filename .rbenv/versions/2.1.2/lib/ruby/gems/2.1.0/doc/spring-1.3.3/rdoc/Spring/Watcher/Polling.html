def start
  unless @poller
    @poller = Thread.new {
      Thread.current.abort_on_exception = true

      loop do
        Kernel.sleep latency
        check_stale
      end
    }
  end
end

def compute_mtime
  expanded_files.map { |f| File.mtime(f).to_f }.max || 0
rescue Errno::ENOENT
  # if a file does no longer exist, the watcher is always stale.
  Float::MAX
end
